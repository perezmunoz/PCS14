% les théorèmes
\newtheorem{thm}{\textsc{Théorème}}
\newcommand{\theoreme}[2]{
  \begin{center}
    \fcolorbox{red}{redclair}{
      \begin{minipage}{0.9\textwidth}
        \begin{thm}[\emph{#1}] \color{red} \ \newline
              #2
        \end{thm}
      \end{minipage}}
  \end{center}}

}
